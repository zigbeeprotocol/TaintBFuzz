\begin{syntax}
  binders ::= binder (, binder)* ;
  \
  binder ::= type-expr variable-ident;
             (,variable-ident)*
  \
  type-expr ::= logic-type-expr | C-type-name
  \
  logic-type-expr ::= built-in-logic-type ;
  | id ; type identifier
  \
  built-in-logic-type ::= "boolean" | "integer" | "real"
  \
  variable-ident ::= id
  | "*" variable-ident ;
  | variable-ident "[]";
  | "(" variable-ident ")"
\end{syntax}
