  { data-inv-def } ::= { data-invariant } | { type-invariant }
  \
  { data-invariant } ::= { { inv-strength? } "global" "invariant" } ;

                      { id "(" C-type-name id ")" "=" pred ";" }

  { inv-strength } ::= { "weak" } | { "strong" }
\end{syntax}
