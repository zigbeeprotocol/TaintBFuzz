\newenvironment{debutenv}[1]
  {
    \vspace{5mm}
    \begin{footnotesize}
      \setenvclass{minipage}{myminipage}
      \noindent\begin{minipage}[t]{\textwidth}
      \fbox{\bf #1} \hrule
    \vspace{3mm}
  }
  {
    \end{minipage}
    \end{footnotesize}
  }
\newenvironment{finenv}
  {
    \begin{footnotesize}
      \setenvclass{minipage}{myminipage}
    \noindent\begin{minipage}[t]{\textwidth}
  }
  {
    \hrule
    \
    \end{minipage}
    \end{footnotesize}
    \vspace{5mm}
  }
\newenvironment{monenv}[1]
  {\begin{debutenv}{#1}}
  {\end{debutenv}
   \par
   \begin{finenv}
   \end{finenv}}
